void reshape(int w, int h) {
    glViewport(0, 0, w, h);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    glOrtho(0, 50, 0, 50, -1, 1);
    glMatrixMode(GL_MODELVIEW);
    glLoadIdentity();
}
          
        
          
void drawScene() {
    glClear(GL_COLOR_BUFFER_BIT);
    glPushMatrix();
        glTranslatef(25, 0, 0);
        glBegin(GL_TRIANGLE_FAN);
            glVertex2f(-0.2, -0.2);
            glVertex2f( 0.2, -0.2);
            glVertex2f( 0.2,  0.2);
            glVertex2f(-0.2,  0.2);
        glEnd();
    glPopMatrix();
    glFlush();
}
